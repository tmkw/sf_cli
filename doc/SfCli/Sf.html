    target_org

    an org alias name, which is used for sf command operations (default is nil). If target_org is nil, the command uses the default org.

def initialize(target_org: nil)
  @org = Org.new(target_org)
  @data = Data.new(target_org)
  @sobject = SObject.new(target_org)
  @project = Project.new
end
