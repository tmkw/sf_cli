def prepare_record_in_bulk_mode(hash) # :doc:
  hash.keys.each do |k|
    match_result = /(.*)\.(.*)/.match(k)
    if match_result
      hash[match_result[1]] = {match_result[2] => hash[k]}
      hash.delete k
    end
  end

  hash
end
